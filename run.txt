// ====== RGB ======
void setRGB(int r, int g, int b) {
  if (commonAnode) {
    r = 255 - r;
    g = 255 - g;
    b = 255 - b;
  }
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}

void rgbOff() {
  setRGB(0, 0, 0);
}

void rgbColorForStep(int step) {
  switch (step % 6) {
    case 0: setRGB(255, 0, 0); break;
    case 1: setRGB(0, 255, 0); break;
    case 2: setRGB(0, 0, 255); break;
    case 3: setRGB(255, 255, 0); break;
    case 4: setRGB(0, 255, 255); break;
    case 5: setRGB(255, 0, 255); break;
  }
}

// ====== ДАТЧИК ======
long readDistanceCM() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH, 30000);
  if (duration == 0) return -1;

  return duration * 0.034 / 2;
}

// ====== LCD АНИМАЦИЯ ======
void loadBarGlyphs() {
  byte g0[8] = {0,0,0,0,0,0,0,0};
  byte g1[8] = {0,0,0,0,0,0,0,31};
  byte g2[8] = {0,0,0,0,0,0,31,31};
  byte g3[8] = {0,0,0,0,0,31,31,31};
  byte g4[8] = {0,0,0,0,31,31,31,31};
  byte g5[8] = {0,0,0,31,31,31,31,31};
  byte g6[8] = {0,0,31,31,31,31,31,31};
  byte g7[8] = {0,31,31,31,31,31,31,31};

  lcd.createChar(0, g0);
  lcd.createChar(1, g1);
  lcd.createChar(2, g2);
  lcd.createChar(3, g3);
  lcd.createChar(4, g4);
  lcd.createChar(5, g5);
  lcd.createChar(6, g6);
  lcd.createChar(7, g7);
}

void drawWaveFrame(byte frame) {
  for (int col = 0; col < 16; col++) {
    byte levelTop = (frame + col) % 8;
    lcd.setCursor(col, 0);
    lcd.write(levelTop);
  }
}

void renderIdleScreen(byte frame) {
  lcd.setCursor(0, 0);
  drawWaveFrame(frame);

  lcd.setCursor(0, 1);
  lcd.print("Waiting...      ");
}

void renderMusicScreen(byte frame) {
  lcd.setCursor(0, 0);
  drawWaveFrame(frame);

  lcd.setCursor(0, 1);
  lcd.print("Playing music   ");
}

void bootScreen() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("SMART DESK");
  lcd.setCursor(0, 1);
  lcd.print("Boot sequence");

  for (int i = 0; i < 12; i++) {
    lcd.setCursor(14, 1);
    lcd.print(">");
    delay(100);
    lcd.setCursor(14, 1);
    lcd.print(" ");
    delay(100);
  }

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Waiting...");
  lcd.setCursor(0, 1);
  lcd.print("Dist: --- cm");
}

// ====== МУЗЫКА + АНИМАЦИЯ ВМЕСТЕ ======
void playThemeWithRGB() {
  lcd.clear();
  lcd.setCursor(0, 1);
  lcd.print("Playing music   ");

  byte frame = 0;

  for (int thisNote = 0; thisNote < notes * 2; thisNote += 2) {
    int divider = melody[thisNote + 1];
    int noteDuration;

    if (divider > 0) {
      noteDuration = wholenote / divider;
    } else {
      noteDuration = wholenote / abs(divider);
      noteDuration = noteDuration * 3 / 2;
    }

    int note = melody[thisNote];
    int soundTime = noteDuration * 9 / 10;
    int restTime = noteDuration - soundTime;

    if (note == REST) {
      rgbOff();
      noTone(buzzerPin);

      unsigned long start = millis();
      while (millis() - start < (unsigned long)noteDuration) {
        renderMusicScreen(frame);
        frame++;
        delay(80);
      }
    } else {
      rgbColorForStep(thisNote / 2);
      tone(buzzerPin, note);

      unsigned long start = millis();
      while (millis() - start < (unsigned long)soundTime) {
        renderMusicScreen(frame);
        frame++;
        delay(60);
      }

      noTone(buzzerPin);
      rgbOff();

      start = millis();
      while (millis() - start < (unsigned long)restTime) {
        renderMusicScreen(frame);
        frame++;
        delay(40);
      }
    }
  }

  noTone(buzzerPin);
  rgbOff();
}

// ====== ПРИВЕТСТВИЕ ======
void welcomeAnimation() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("WELCOME BACK");
  lcd.setCursor(0, 1);
  lcd.print("Smart Desk Ready");

  for (int i = 0; i < 6; i++) {
    setRGB(255, 0, 100);
    delay(120);
    setRGB(0, 255, 255);
    delay(120);
  }

  rgbOff();
  delay(1000);
}

// ====== SETUP ======
void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzerPin, OUTPUT);

  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);

  lcd.begin(16, 2);
  loadBarGlyphs();
  bootScreen();

  setRGB(0, 0, 255);
  delay(150);
  setRGB(0, 255, 0);
  delay(150);
  setRGB(255, 0, 0);
  delay(150);
  rgbOff();
}

// ====== LOOP ======
void loop() {
  long d = readDistanceCM();

  if (d != -1 && d < triggerDistanceCm && !triggered) {
    triggered = true;
    playThemeWithRGB();
    welcomeAnimation();
  }

  if (d != -1 && d > resetDistanceCm) {
    triggered = false;
  }

  if (!triggered) {
    static byte frame = 0;
    static unsigned long lastFrame = 0;

    if (millis() - lastFrame >= 120) {
      lastFrame = millis();
      renderIdleScreen(frame);
      frame++;
    }

    if (d != -1 && d < resetDistanceCm) {
      setRGB(0, 0, 20);
    } else {
      rgbOff();
    }
  }

  delay(20);
}
