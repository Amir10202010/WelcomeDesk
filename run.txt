#include <LiquidCrystal.h>

// ===================== LCD =====================
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

// ===================== IMPORTANT =====================
// Buttons must be wired from pin -> button -> GND
// because INPUT_PULLUP is used.

// If your RGB LED is common anode, set true.
const bool commonAnode = false;

// ===================== PINS =====================
const int trigPin = 22;
const int echoPin = 23;

const int buzzerPin = 8;

const int redPin   = 6;
const int greenPin = 7;
const int bluePin  = 10;

const int modeButtonPin   = 24;   // MODE / next menu
const int actionButtonPin  = 25;   // ACTION / play / reset

const int joyXPin = A1;
const int joyYPin = A2;
const int potPin  = A0;

// Raw 8x8 matrix wiring assumed here.
// If your matrix is MAX7219-based, this part must be changed.
const byte rowPins[8] = {30, 31, 32, 33, 34, 35, 36, 37};
const byte colPins[8] = {38, 39, 40, 41, 42, 43, 44, 45};

// ===================== MODES =====================
enum Mode {
  MODE_DISTANCE = 0,
  MODE_MUSIC    = 1,
  MODE_GAME     = 2
};

Mode currentMode = MODE_DISTANCE;

// ===================== BUTTON DEBOUNCE =====================
struct ButtonState {
  bool stable = HIGH;
  bool lastReading = HIGH;
  unsigned long lastChange = 0;
};

ButtonState modeBtn;
ButtonState actionBtn;

const unsigned long debounceMs = 40;

// ===================== LCD HELPERS =====================
void printLine(byte row, const char *text) {
  char buf[17];
  snprintf(buf, sizeof(buf), "%-16.16s", text);
  lcd.setCursor(0, row);
  lcd.print(buf);
}

// ===================== RGB HELPERS =====================
void setRGB(int r, int g, int b) {
  if (commonAnode) {
    r = 255 - r;
    g = 255 - g;
    b = 255 - b;
  }
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}

void rgbOff() {
  setRGB(0, 0, 0);
}

void rgbColorForStep(int step) {
  switch (step % 6) {
    case 0: setRGB(255, 0, 0); break;
    case 1: setRGB(0, 255, 0); break;
    case 2: setRGB(0, 0, 255); break;
    case 3: setRGB(255, 255, 0); break;
    case 4: setRGB(0, 255, 255); break;
    default:setRGB(255, 0, 255); break;
  }
}

// ===================== SENSOR =====================
long readDistanceCM() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  unsigned long duration = pulseIn(echoPin, HIGH, 30000UL);
  if (duration == 0) return -1;

  return (long)(duration * 0.034 / 2.0);
}

// ===================== BUTTON READ =====================
bool buttonPressed(uint8_t pin, ButtonState &btn) {
  bool reading = digitalRead(pin);

  if (reading != btn.lastReading) {
    btn.lastChange = millis();
    btn.lastReading = reading;
  }

  if ((millis() - btn.lastChange) >= debounceMs && reading != btn.stable) {
    btn.stable = reading;
    if (btn.stable == LOW) return true;   // press event
  }

  return false;
}

// ===================== MATH / MODE =====================
Mode nextModeOf(Mode m) {
  if (m == MODE_DISTANCE) return MODE_MUSIC;
  if (m == MODE_MUSIC)    return MODE_GAME;
  return MODE_DISTANCE;
}

// ===================== BOOT =====================
void bootScreen() {
  lcd.clear();
  printLine(0, "SMART WORK DESK");
  printLine(1, "Booting...");

  for (byte i = 0; i < 10; i++) {
    lcd.setCursor(14, 1);
    lcd.print((i % 2 == 0) ? ">" : " ");
    delay(90);
  }

  lcd.clear();
}

// ===================== MUSIC =====================
// Simple but pleasant tune.
// Format: {NOTE, DIVIDER}
#define REST 0
#define NOTE_C4 262
#define NOTE_D4 294
#define NOTE_E4 330
#define NOTE_F4 349
#define NOTE_G4 392
#define NOTE_A4 440
#define NOTE_B4 494
#define NOTE_C5 523
#define NOTE_D5 587
#define NOTE_E5 659
#define NOTE_F5 698
#define NOTE_G5 784

const int melody[][2] = {
  {NOTE_E4, 8}, {NOTE_G4, 8}, {NOTE_A4, 8}, {REST, 8},
  {NOTE_A4, 8}, {NOTE_C5, 8}, {NOTE_B4, 8}, {REST, 8},
  {NOTE_A4, 8}, {NOTE_G4, 8}, {NOTE_E4, 4}, {REST, 4},
  {NOTE_E4, 8}, {NOTE_G4, 8}, {NOTE_A4, 8}, {NOTE_B4, 8},
  {NOTE_C5, 4}, {REST, 4}
};

const int melodyLen = sizeof(melody) / sizeof(melody[0]);
const int tempo = 132;
const unsigned long wholenote = (60000UL * 4) / tempo;

bool musicPlaying = false;
int musicIndex = 0;
bool musicSoundPhase = true;
unsigned long musicPhaseEnd = 0;
unsigned long lastMusicLcd = 0;
byte musicSpinner = 0;

void stopMusic() {
  musicPlaying = false;
  noTone(buzzerPin);
  rgbOff();
}

void startMusic() {
  musicPlaying = true;
  musicIndex = 0;
  musicSoundPhase = true;
  musicPhaseEnd = 0;
  lastMusicLcd = 0;
  musicSpinner = 0;
  lcd.clear();
}

void toggleMusic() {
  if (musicPlaying) stopMusic();
  else startMusic();
}

void updateMusicPlayer() {
  unsigned long now = millis();

  if (musicIndex >= melodyLen) {
    stopMusic();
    printLine(0, "MODE: MUSIC");
    printLine(1, "DONE  A=PLAY");
    return;
  }

  if (now >= musicPhaseEnd) {
    int note = melody[musicIndex][0];
    int divider = melody[musicIndex][1];

    unsigned long noteDuration;
    if (divider > 0) noteDuration = wholenote / divider;
    else noteDuration = (wholenote / abs(divider)) * 3UL / 2UL;

    if (musicSoundPhase) {
      if (note == REST) {
        noTone(buzzerPin);
        rgbOff();
        musicPhaseEnd = now + noteDuration;
        musicSoundPhase = true;
        musicIndex++;
      } else {
        tone(buzzerPin, note);
        rgbColorForStep(musicIndex);
        musicPhaseEnd = now + (noteDuration * 9UL / 10UL);
        musicSoundPhase = false;
      }
    } else {
      noTone(buzzerPin);
      rgbOff();
      musicPhaseEnd = now + (noteDuration / 10UL);
      musicSoundPhase = true;
      musicIndex++;
    }
  }

  if (now - lastMusicLcd >= 120) {
    lastMusicLcd = now;

    char line1[17];
    char bar[11];
    byte percent = (byte)((musicIndex * 100UL) / melodyLen);
    byte filled = percent / 10;

    for (byte i = 0; i < 10; i++) bar[i] = (i < filled) ? '=' : ' ';
    bar[10] = '\0';

    const char spinChars[] = {'|', '/', '-', '\\'};
    char sp = spinChars[musicSpinner++ % 4];

    snprintf(line1, sizeof(line1), "PLAY[%s] %3u%c", bar, percent, sp);
    printLine(0, "MODE: MUSIC");
    printLine(1, line1);
  }
}

void updateMusicIdleScreen() {
  static unsigned long lastIdle = 0;
  static byte anim = 0;
  unsigned long now = millis();

  if (now - lastIdle >= 140) {
    lastIdle = now;
    anim = (anim + 1) % 4;
  }

  const char spinChars[] = {'|', '/', '-', '\\'};
  char line1[17];
  snprintf(line1, sizeof(line1), "A=PLAY     %c", spinChars[anim]);

  printLine(0, "MODE: MUSIC");
  printLine(1, line1);
  setRGB(0, 0, 20);
}

// ===================== 8x8 MATRIX GAME =====================
byte gameFrame[8];
byte matrixRow = 0;
unsigned long lastMatrixMux = 0;

struct Obstacle {
  int8_t x = -1;
  int8_t y = -1;
  bool active = false;
};

const byte maxObstacles = 4;
Obstacle obstacles[maxObstacles];

int playerX = 3;
int lives = 3;
unsigned long score = 0;
bool gameOver = false;

unsigned long lastGameStep = 0;
unsigned long lastSpawn = 0;
unsigned long lastJoyMove = 0;
unsigned long lastGameLcd = 0;

void clearGameFrame() {
  for (byte r = 0; r < 8; r++) gameFrame[r] = 0;
}

void setFramePixel(int x, int y, bool on) {
  if (x < 0 || x > 7 || y < 0 || y > 7) return;
  if (on) gameFrame[y] |= (1 << (7 - x));
  else    gameFrame[y] &= ~(1 << (7 - x));
}

void setupMatrixPins() {
  for (byte i = 0; i < 8; i++) {
    pinMode(rowPins[i], OUTPUT);
    pinMode(colPins[i], OUTPUT);
    digitalWrite(rowPins[i], LOW);
    digitalWrite(colPins[i], HIGH); // off by default
  }
}

void serviceMatrix() {
  if (millis() - lastMatrixMux < 2) return;
  lastMatrixMux = millis();

  // turn off previous row
  digitalWrite(rowPins[matrixRow], LOW);

  // next row
  matrixRow = (matrixRow + 1) & 0x07;

  // set columns for that row
  byte rowBits = gameFrame[matrixRow];
  for (byte c = 0; c < 8; c++) {
    bool on = rowBits & (1 << (7 - c));
    digitalWrite(colPins[c], on ? LOW : HIGH);  // swap if your matrix is inverted
  }

  // turn row on
  digitalWrite(rowPins[matrixRow], HIGH);
}

void resetGame() {
  clearGameFrame();

  for (byte i = 0; i < maxObstacles; i++) {
    obstacles[i].active = false;
    obstacles[i].x = -1;
    obstacles[i].y = -1;
  }

  playerX = 3;
  lives = 3;
  score = 0;
  gameOver = false;
  lastGameStep = 0;
  lastSpawn = 0;
  lastJoyMove = 0;
  lastGameLcd = 0;

  lcd.clear();
  printLine(0, "MODE: GAME");
  printLine(1, "A=RESET");
}

void spawnObstacle() {
  for (byte i = 0; i < maxObstacles; i++) {
    if (!obstacles[i].active) {
      obstacles[i].active = true;
      obstacles[i].x = random(0, 8);
      obstacles[i].y = 0;
      return;
    }
  }
}

unsigned long speedFromPot() {
  int pot = analogRead(potPin);
  return map(pot, 0, 1023, 450, 130);   // higher pot = faster
}

unsigned long spawnFromPot() {
  int pot = analogRead(potPin);
  return map(pot, 0, 1023, 1200, 400);  // higher pot = more frequent
}

void buildGameFrame() {
  clearGameFrame();

  // obstacles
  for (byte i = 0; i < maxObstacles; i++) {
    if (obstacles[i].active) {
      setFramePixel(obstacles[i].x, obstacles[i].y, true);
    }
  }

  // player
  setFramePixel(playerX, 7, true);
}

void updateGameMode() {
  unsigned long now = millis();

  if (gameOver) {
    printLine(0, "GAME OVER");
    printLine(1, "A=RESET");
    setRGB(80, 0, 0);
    clearGameFrame();
    setFramePixel(3, 3, true);
    setFramePixel(4, 3, true);
    setFramePixel(3, 4, true);
    setFramePixel(4, 4, true);
    return;
  }

  // joystick move
  if (now - lastJoyMove >= 90) {
    lastJoyMove = now;
    int joyX = analogRead(joyXPin);

    if (joyX < 350 && playerX > 0) playerX--;
    if (joyX > 650 && playerX < 7) playerX++;
  }

  unsigned long stepInterval = speedFromPot();
  unsigned long spawnInterval = spawnFromPot();

  // spawn obstacles
  if (now - lastSpawn >= spawnInterval) {
    lastSpawn = now;
    spawnObstacle();
  }

  // move obstacles
  if (now - lastGameStep >= stepInterval) {
    lastGameStep = now;

    for (byte i = 0; i < maxObstacles; i++) {
      if (!obstacles[i].active) continue;

      obstacles[i].y++;

      if (obstacles[i].y >= 7) {
        if (obstacles[i].x == playerX) {
          lives--;
          if (lives <= 0) {
            gameOver = true;
          }
        } else {
          score++;
        }
        obstacles[i].active = false;
      }
    }
  }

  buildGameFrame();

  // RGB feedback
  if (lives == 3) setRGB(0, 30, 0);
  else if (lives == 2) setRGB(30, 20, 0);
  else setRGB(40, 0, 0);

  // LCD update
  if (now - lastGameLcd >= 140) {
    lastGameLcd = now;

    char line0[17];
    char line1[17];
    unsigned long shownScore = score;

    snprintf(line0, sizeof(line0), "GAME S:%lu", shownScore);
    snprintf(line1, sizeof(line1), "L:%d SP:%lu", lives, stepInterval);

    printLine(0, line0);
    printLine(1, line1);
  }
}

// ===================== DISTANCE MODE =====================
unsigned long lastDistanceUpdate = 0;
long lastDistance = -1;

void updateDistanceMode() {
  unsigned long now = millis();

  if (now - lastDistanceUpdate >= 120) {
    lastDistanceUpdate = now;
    lastDistance = readDistanceCM();
  }

  char line1[17];
  printLine(0, "MODE: DISTANCE");

  if (lastDistance < 0) {
    snprintf(line1, sizeof(line1), "Dist: --- cm");
    rgbOff();
  } else {
    if (lastDistance < 20) {
      setRGB(255, 80, 0);
    } else if (lastDistance < 50) {
      setRGB(0, 0, 40);
    } else {
      rgbOff();
    }

    snprintf(line1, sizeof(line1), "Dist: %3ld cm", lastDistance);
  }

  printLine(1, line1);
}

// ===================== MODE SWITCH =====================
void enterMode(Mode newMode) {
  currentMode = newMode;
  stopMusic();
  clearGameFrame();
  lcd.clear();

  if (currentMode == MODE_DISTANCE) {
    printLine(0, "MODE: DISTANCE");
    printLine(1, "Ready");
  } else if (currentMode == MODE_MUSIC) {
    printLine(0, "MODE: MUSIC");
    printLine(1, "A=PLAY");
  } else {
    resetGame();
  }
}

// ===================== BUTTON HANDLER =====================
void handleButtons() {
  if (buttonPressed(modeButtonPin, modeBtn)) {
    enterMode(nextModeOf(currentMode));
  }

  if (buttonPressed(actionButtonPin, actionBtn)) {
    if (currentMode == MODE_MUSIC) {
      toggleMusic();
      if (!musicPlaying) {
        printLine(0, "MODE: MUSIC");
        printLine(1, "A=PLAY");
      }
    } else if (currentMode == MODE_GAME) {
      if (gameOver) {
        resetGame();
      }
    }
  }
}

// ===================== SETUP =====================
void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  pinMode(buzzerPin, OUTPUT);

  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);

  pinMode(modeButtonPin, INPUT_PULLUP);
  pinMode(actionButtonPin, INPUT_PULLUP);

  lcd.begin(16, 2);
  lcd.clear();

  setupMatrixPins();

  randomSeed(analogRead(A15) ^ micros());

  bootScreen();
  enterMode(MODE_DISTANCE);

  // startup RGB animation
  setRGB(0, 0, 255); delay(110);
  setRGB(0, 255, 0); delay(110);
  setRGB(255, 0, 0); delay(110);
  rgbOff();
}

// ===================== LOOP =====================
void loop() {
  handleButtons();

  if (currentMode == MODE_DISTANCE) {
    updateDistanceMode();
  } else if (currentMode == MODE_MUSIC) {
    if (musicPlaying) updateMusicPlayer();
    else updateMusicIdleScreen();
  } else if (currentMode == MODE_GAME) {
    updateGameMode();
  }

  serviceMatrix();
}
