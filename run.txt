// ====== RGB ======
void setRGB(int r, int g, int b) {
  if (commonAnode) {
    r = 255 - r;
    g = 255 - g;
    b = 255 - b;
  }
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}

void rgbOff() {
  setRGB(0, 0, 0);
}

void rgbPulse(int r, int g, int b, int stepsDelay = 4) {
  for (int x = 0; x <= 255; x += 8) {
    setRGB((r * x) / 255, (g * x) / 255, (b * x) / 255);
    delay(stepsDelay);
  }
  for (int x = 255; x >= 0; x -= 8) {
    setRGB((r * x) / 255, (g * x) / 255, (b * x) / 255);
    delay(stepsDelay);
  }
}

void colorStep(int step) {
  switch (step % 6) {
    case 0: setRGB(255, 0, 0); break;
    case 1: setRGB(0, 255, 0); break;
    case 2: setRGB(0, 0, 255); break;
    case 3: setRGB(255, 255, 0); break;
    case 4: setRGB(0, 255, 255); break;
    case 5: setRGB(255, 0, 255); break;
  }
}

// ====== DISTANCE ======
long readDistanceCM() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);

  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);

  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH, 30000);
  if (duration == 0) return -1;

  return duration * 0.034 / 2;
}

// ====== START SCREEN ======
void bootScreen() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("SMART DESK");
  lcd.setCursor(0, 1);
  lcd.print("Booting...");

  for (int t = 0; t < 10; t++) {
    setRGB(0, 0, 255);
    delay(60);
    setRGB(0, 255, 0);
    delay(60);
  }
  rgbOff();

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Waiting for you");
  lcd.setCursor(0, 1);
  lcd.print("Dist: --- cm");
}

// ====== MUSIC + RGB ======
void playThemeWithRGB() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Playing theme");
  lcd.setCursor(0, 1);
  lcd.print("Please wait...");

  for (int thisNote = 0; thisNote < notes * 2; thisNote += 2) {
    int divider = melody[thisNote + 1];
    int noteDuration;

    if (divider > 0) {
      noteDuration = wholenote / divider;
    } else {
      noteDuration = wholenote / abs(divider);
      noteDuration = noteDuration * 3 / 2;
    }

    int note = melody[thisNote];

    if (note == REST) {
      rgbOff();
      noTone(buzzerPin);
      delay(noteDuration);
    } else {
      colorStep(thisNote / 2);
      tone(buzzerPin, note, noteDuration * 9 / 10);
      delay(noteDuration * 9 / 10);
      noTone(buzzerPin);
      rgbOff();
      delay(noteDuration / 10);
    }
  }

  noTone(buzzerPin);
  rgbOff();
}

// ====== FINAL WELCOME ======
void finalWelcome() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Smart Desk Ready");
  lcd.setCursor(0, 1);
  lcd.print("Have a nice day");

  for (int x = 0; x < 4; x++) {
    setRGB(255, 0, 120);
    delay(120);
    setRGB(0, 255, 255);
    delay(120);
  }
  rgbOff();
}

// ====== FULL INTRO SEQUENCE ======
void runIntroSequence() {
  teachMeAnimation();
  playThemeWithRGB();
  finalWelcome();
}

// ====== SETUP ======
void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzerPin, OUTPUT);

  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);

  randomSeed(analogRead(A0));

  lcd.init();
  lcd.backlight();

  lcd.createChar(4, customChar1);
  lcd.createChar(3, customChar2);
  lcd.createChar(2, customChar3);
  lcd.createChar(1, customChar4);
  lcd.createChar(0, customChar5);

  bootScreen();
}

// ====== LOOP ======
void loop() {
  long d = readDistanceCM();

  lcd.setCursor(0, 1);
  if (d == -1) {
    lcd.print("Dist: --- cm   ");
  } else {
    lcd.print("Dist: ");
    lcd.print(d);
    lcd.print(" cm   ");
  }

  if (d != -1 && d < triggerDistanceCm && !triggered) {
    triggered = true;
    runIntroSequence();
  }

  if (d != -1 && d > resetDistanceCm) {
    triggered = false;
  }

  if (!triggered && d != -1 && d < resetDistanceCm) {
    setRGB(0, 0, 20);
  } else if (!triggered) {
    rgbOff();
  }

  delay(120);
}
