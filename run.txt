// ===================== RGB HELPERS =====================
void setRGB(int r, int g, int b) {
  if (commonAnode) {
    r = 255 - r;
    g = 255 - g;
    b = 255 - b;
  }
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}

void rgbOff() {
  setRGB(0, 0, 0);
}

void rgbColorForStep(int step) {
  switch (step % 6) {
    case 0: setRGB(255, 0, 0); break;
    case 1: setRGB(0, 255, 0); break;
    case 2: setRGB(0, 0, 255); break;
    case 3: setRGB(255, 255, 0); break;
    case 4: setRGB(0, 255, 255); break;
    case 5: setRGB(255, 0, 255); break;
  }
}

// ===================== SENSOR =====================
long readDistanceCM() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH, 30000);
  if (duration == 0) return -1;

  return duration * 0.034 / 2;
}

// ===================== LCD ANIMATION =====================
void loadBarGlyphs() {
  byte g0[8] = {0,0,0,0,0,0,0,0};
  byte g1[8] = {0,0,0,0,0,0,0,31};
  byte g2[8] = {0,0,0,0,0,0,31,31};
  byte g3[8] = {0,0,0,0,0,31,31,31};
  byte g4[8] = {0,0,0,0,31,31,31,31};
  byte g5[8] = {0,0,0,31,31,31,31,31};
  byte g6[8] = {0,0,31,31,31,31,31,31};
  byte g7[8] = {0,31,31,31,31,31,31,31};

  lcd.createChar(0, g0);

  lcd.createChar(2, g2);

  lcd.createChar(4, g4);

  lcd.createChar(7, g7);

void drawWaveFrame(byte frame) {
  for (int col = 0; col < 16; col++) {
    byte levelTop = (frame + col) % 8;
    lcd.setCursor(col, 0);
    lcd.write(levelTop);
  }
}

void renderIdleScreen(byte frame) {
  drawWaveFrame(frame);
  lcd.setCursor(0, 1);
  lcd.print("Waiting...      ");
}

void renderMusicScreen(byte frame) {
  drawWaveFrame(frame);
  lcd.setCursor(0, 1);
  lcd.print("Music mode      ");
}

void renderGameScreen(byte frame) {
  drawWaveFrame(frame);
  lcd.setCursor(0, 1);
  lcd.print("Game: Dodge Duel");
}

void bootScreen() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("SMART DESK");
  lcd.setCursor(0, 1);
  lcd.print("Boot sequence");

  for (int i = 0; i < 12; i++) {
    lcd.setCursor(14, 1);
    lcd.print(">");
    delay(100);
    lcd.setCursor(14, 1);
    lcd.print(" ");
    delay(100);
  }

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Waiting...");
  lcd.setCursor(0, 1);
  lcd.print("Dist: --- cm");
}

// ===================== MUSIC =====================
void playThemeWithRGB() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Playing theme");
  lcd.setCursor(0, 1);
  lcd.print("Please wait...");

  byte frame = 0;

  for (int thisNote = 0; thisNote < notes * 2; thisNote += 2) {
    int divider = melody[thisNote + 1];
    int noteDuration;

    if (divider > 0) {
      noteDuration = wholenote / divider;
    } else {
      noteDuration = wholenote / abs(divider);
      noteDuration = noteDuration * 3 / 2;
    }

    int note = melody[thisNote];
    int soundTime = noteDuration * 9 / 10;
    int restTime  = noteDuration - soundTime;

    if (note == REST) {
      rgbOff();
      noTone(buzzerPin);

      unsigned long start = millis();
      while (millis() - start < (unsigned long)noteDuration) {
        renderMusicScreen(frame);
        frame++;
        delay(60);
      }
    } else {
      rgbColorForStep(thisNote / 2);
      tone(buzzerPin, note);

      unsigned long start = millis();
      while (millis() - start < (unsigned long)soundTime) {
        renderMusicScreen(frame);
        frame++;
        delay(60);
      }

      noTone(buzzerPin);
      rgbOff();

      start = millis();
      while (millis() - start < (unsigned long)restTime) {
        renderMusicScreen(frame);
        frame++;
        delay(40);
      }
    }
  }

  noTone(buzzerPin);
  rgbOff();
}

void gameScreenIdle() {
  static byte frame = 0;
  static unsigned long lastFrame = 0;

  if (millis() - lastFrame >= 120) {
    lastFrame = millis();
    renderGameScreen(frame);
    frame++;
  }

  int joyX = analogRead(joyXPin);
  int joyY = analogRead(joyYPin);
  int potV = analogRead(potPin);

  lcd.setCursor(0, 1);
  char line[17];
  snprintf(line, sizeof(line), "J:%4d %4d   ", joyX, joyY);
  lcd.print(line);

  // Pot is reserved for difficulty later
  (void)potV;
}

// ===================== DISTANCE MODE =====================
void distanceScreen() {
  static byte frame = 0;
  static unsigned long lastFrame = 0;

  long d = readDistanceCM();

  if (millis() - lastFrame >= 120) {
    lastFrame = millis();
    renderIdleScreen(frame);
    frame++;
  }

  lcd.setCursor(0, 1);
  char line[17];
  if (d == -1) {
    snprintf(line, sizeof(line), "Dist: --- cm   ");
  } else {
    snprintf(line, sizeof(line), "Dist:%3ld cm    ", d);
  }
  lcd.print(line);

  if (d != -1 && d < 30) {
    setRGB(255, 120, 0);
  } else if (d != -1 && d < 70) {
    setRGB(0, 0, 25);
  } else {
    rgbOff();
  }
}

// ===================== MENU =====================
void showModeHeader() {
  lcd.setCursor(0, 0);
  switch (currentMode) {
    case MODE_DISTANCE:
      lcd.print("MODE: DISTANCE ");
      break;
    case MODE_MUSIC:
      lcd.print("MODE: MUSIC    ");
      break;
    case MODE_GAME:
      lcd.print("MODE: GAME     ");
      break;
  }
}

void nextMode() {
  if (currentMode == MODE_DISTANCE) currentMode = MODE_MUSIC;
  else if (currentMode == MODE_MUSIC) currentMode = MODE_GAME;
  else currentMode = MODE_DISTANCE;

  lcd.clear();
  showModeHeader();
}

bool buttonPressed(int pin, bool &lastState, unsigned long &lastDebounce) {
  bool reading = digitalRead(pin);

  if (reading != lastState) {
    lastDebounce = millis();
  }

  if ((millis() - lastDebounce) > debounceMs) {
    if (lastState == HIGH && reading == LOW) {
      lastState = reading;
      return true;
    }
  }

  lastState = reading;
  return false;
}

// ===================== SETUP =====================
void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  pinMode(buzzerPin, OUTPUT);

  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);

  pinMode(modeButtonPin, INPUT_PULLUP);
  pinMode(actionButtonPin, INPUT_PULLUP);

  lcd.begin(16, 2);
  loadBarGlyphs();
  bootScreen();

  setRGB(0, 0, 255);
  delay(150);
  setRGB(0, 255, 0);
  delay(150);
  setRGB(255, 0, 0);
  delay(150);
  rgbOff();
}

// ===================== LOOP =====================
void loop() {
  if (buttonPressed(modeButtonPin, lastModeBtn, lastModeDebounce)) {
    nextMode();
  }

  if (currentMode == MODE_DISTANCE) {
    showModeHeader();
    distanceScreen();
  }

  else if (currentMode == MODE_MUSIC) {
    showModeHeader();
    renderMusicScreen((byte)(millis() / 120));

    lcd.setCursor(0, 1);
    lcd.print("A=PLAY MODE     ");

    if (buttonPressed(actionButtonPin, lastActionBtn, lastActionDebounce)) {
      playThemeWithRGB();
      lcd.clear();
      showModeHeader();
    }

    // small idle glow
    setRGB(0, 0, 20);
  }

  else if (currentMode == MODE_GAME) {
    showModeHeader();
    gameScreenIdle();

    lcd.setCursor(0, 1);
    lcd.print("A=START GAME    ");

    if (buttonPressed(actionButtonPin, lastActionBtn, lastActionDebounce)) {
      startDodgeDuel();
      lcd.clear();
      showModeHeader();
    }

    // simple idle color for game mode
    setRGB(20, 0, 20);
  }

  delay(20);
}
