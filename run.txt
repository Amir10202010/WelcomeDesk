// ===================== RGB HELPERS =====================
void setRGB(int r, int g, int b) {
  if (commonAnode) {
    r = 255 - r;
    g = 255 - g;
    b = 255 - b;
  }
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}

void rgbOff() {
  setRGB(0, 0, 0);
}

void rgbColorForStep(int step) {
  switch (step % 6) {
    case 0: setRGB(255, 0, 0); break;
    case 1: setRGB(0, 255, 0); break;
    case 2: setRGB(0, 0, 255); break;
    case 3: setRGB(255, 255, 0); break;
    case 4: setRGB(0, 255, 255); break;
    case 5: setRGB(255, 0, 255); break;
  }
}

// ===================== SMALL LCD HELPERS =====================
void printLine(uint8_t row, const char *text) {
  char buf[17];
  snprintf(buf, sizeof(buf), "%-16.16s", text);
  lcd.setCursor(0, row);
  lcd.print(buf);
}

void makeStatusLine(char *out, const char *left, const char spinner) {
  char buf[17];
  snprintf(buf, sizeof(buf), "%-13.13s %c", left, spinner);
  snprintf(out, 17, "%-16.16s", buf);
}

// ===================== SENSOR =====================
long readDistanceCM() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH, 30000);
  if (duration == 0) return -1;

  return duration * 0.034 / 2;
}

// ===================== BOOT =====================
void bootScreen() {
  lcd.clear();
  printLine(0, "SMART DESK");
  printLine(1, "Boot sequence");

  for (int i = 0; i < 12; i++) {
    lcd.setCursor(14, 1);
    lcd.print(">");
    delay(90);
    lcd.setCursor(14, 1);
    lcd.print(" ");
    delay(90);
  }

  lcd.clear();
  printLine(0, "MODE: DISTANCE");
  printLine(1, "Waiting...");
}

// ===================== BUTTONS =====================
bool buttonPressed(uint8_t pin, ButtonState &btn) {
  bool reading = digitalRead(pin);

  if (reading != btn.lastReading) {
    btn.lastChange = millis();
    btn.lastReading = reading;
  }

  if ((millis() - btn.lastChange) > debounceMs) {
    if (reading != btn.stable) {
      btn.stable = reading;
      if (btn.stable == LOW) return true;   // pressed
    }
  }

  return false;
}

void nextMode() {
  if (currentMode == MODE_DISTANCE) currentMode = MODE_MUSIC;
  else if (currentMode == MODE_MUSIC) currentMode = MODE_GAME;
  else currentMode = MODE_DISTANCE;

  lcd.clear();

  if (currentMode == MODE_DISTANCE) printLine(0, "MODE: DISTANCE");
  if (currentMode == MODE_MUSIC)    printLine(0, "MODE: MUSIC");
  if (currentMode == MODE_GAME)     printLine(0, "MODE: GAME");

  printLine(1, " ");
}

// ===================== MUSIC PLAYER STATE =====================
bool musicPlaying = false;
int musicIndex = 0;
bool musicNeedSoundPhase = true;
unsigned long musicPhaseEnd = 0;
int currentNoteDuration = 0;
byte musicFrame = 0;
unsigned long lastMusicFrame = 0;
byte musicSpinner = 0;

void startMusic() {
  musicPlaying = true;
  musicIndex = 0;
  musicNeedSoundPhase = true;
  musicPhaseEnd = 0;
  currentNoteDuration = 0;
  musicFrame = 0;
  musicSpinner = 0;
  noTone(buzzerPin);
  rgbOff();
  lcd.clear();
  printLine(0, "MODE: MUSIC");
  printLine(1, "Starting...");
}

void stopMusic() {
  musicPlaying = false;
  noTone(buzzerPin);
  rgbOff();
}

void updateMusicPlayer() {
  if (!musicPlaying) return;

  if (musicIndex >= notes * 2) {
    stopMusic();
    printLine(0, "MODE: MUSIC");
    printLine(1, "Done");
    return;
  }

  if (millis() - lastMusicFrame >= 80) {
    lastMusicFrame = millis();

    char line1[17];
    byte percent = (byte)((musicIndex * 100UL) / (notes * 2));
    byte bars = percent / 10;

    char bar[11];
    for (int i = 0; i < 10; i++) bar[i] = (i < bars) ? '=' : ' ';
    bar[10] = '\0';

    const char spinChars[] = {'|', '/', '-', '\\'};
    char spinner = spinChars[musicSpinner % 4];
    musicSpinner++;

    snprintf(line1, sizeof(line1), "PLAY[%s] %3u%c", bar, percent, spinner);
    printLine(1, line1);
  }

  if (millis() < musicPhaseEnd) return;

  int note = melody[musicIndex];
  int divider = melody[musicIndex + 1];

  int noteDuration;
  if (divider > 0) noteDuration = wholenote / divider;
  else {
    noteDuration = wholenote / abs(divider);
    noteDuration = noteDuration * 3 / 2;
  }

  if (musicNeedSoundPhase) {
    if (note == REST) {
      noTone(buzzerPin);
      rgbOff();
      musicPhaseEnd = millis() + noteDuration;
      musicIndex += 2;
      return;
    }

    currentNoteDuration = noteDuration;
    tone(buzzerPin, note);
    rgbColorForStep(musicIndex / 2);

    musicPhaseEnd = millis() + (noteDuration * 9L / 10L);
    musicNeedSoundPhase = false;
  } else {
    noTone(buzzerPin);
    rgbOff();

    musicPhaseEnd = millis() + (currentNoteDuration / 10);
    musicNeedSoundPhase = true;
    musicIndex += 2;
  }
}

// ===================== GAME SKELETON =====================
void startDodgeDuel() {
  lcd.clear();
  printLine(0, "Dodge Duel");
  printLine(1, "Coming next...");

  setRGB(0, 255, 255);
  delay(400);
  setRGB(255, 0, 255);
  delay(400);
  rgbOff();
}

void updateGameScreen() {
  static byte anim = 0;
  static unsigned long lastFrame = 0;

  if (millis() - lastFrame >= 120) {
    lastFrame = millis();
    anim = (anim + 1) % 4;
  }

  int joyX = analogRead(joyXPin);
  int joyY = analogRead(joyYPin);
  int potV = analogRead(potPin);

  printLine(0, "MODE: GAME");

  char line1[17];
  const char spinChars[] = {'|', '/', '-', '\\'};
  snprintf(line1, sizeof(line1), "J%4d%4d %c", joyX, joyY, spinChars[anim]);
  printLine(1, line1);

  (void)potV;
}

// ===================== DISTANCE MODE =====================
void updateDistanceScreen() {
  static byte anim = 0;
  static unsigned long lastFrame = 0;

  if (millis() - lastFrame >= 120) {
    lastFrame = millis();
    anim = (anim + 1) % 4;
  }

  long d = readDistanceCM();

  printLine(0, "MODE: DISTANCE");

  char line1[17];
  const char spinChars[] = {'|', '/', '-', '\\'};

  if (d == -1) {
    snprintf(line1, sizeof(line1), "Dist: --- cm  %c", spinChars[anim]);
  } else {
    snprintf(line1, sizeof(line1), "Dist:%3ld cm   %c", d, spinChars[anim]);
  }
  printLine(1, line1);

  if (d != -1 && d < 30) {
    setRGB(255, 120, 0);
  } else if (d != -1 && d < 70) {
    setRGB(0, 0, 25);
  } else {
    rgbOff();
  }
}

// ===================== MUSIC MODE IDLE =====================
void updateMusicIdleScreen() {
  static byte anim = 0;
  static unsigned long lastFrame = 0;

  if (millis() - lastFrame >= 140) {
    lastFrame = millis();
    anim = (anim + 1) % 4;
  }

  printLine(0, "MODE: MUSIC");

  char line1[17];
  const char spinChars[] = {'|', '/', '-', '\\'};

  if (!musicPlaying) {
    snprintf(line1, sizeof(line1), "A=PLAY     %c", spinChars[anim]);
    printLine(1, line1);
    setRGB(0, 0, 20);
  } else {
    updateMusicPlayer();
  }
}

// ===================== SETUP =====================
void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  pinMode(buzzerPin, OUTPUT);

  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);

  pinMode(modeButtonPin, INPUT_PULLUP);
  pinMode(actionButtonPin, INPUT_PULLUP);

  lcd.begin(16, 2);

  bootScreen();

  setRGB(0, 0, 255);
  delay(120);
  setRGB(0, 255, 0);
  delay(120);
  setRGB(255, 0, 0);
  delay(120);
  rgbOff();
}

// ===================== LOOP =====================
void loop() {
  if (buttonPressed(modeButtonPin, modeBtn)) {
    stopMusic();   // stop music if you leave the mode
    nextMode();
  }

  if (currentMode == MODE_DISTANCE) {
    updateDistanceScreen();
  }
  else if (currentMode == MODE_MUSIC) {
    if (buttonPressed(actionButtonPin, actionBtn) && !musicPlaying) {
      startMusic();
    }
    updateMusicIdleScreen();
  }
  else if (currentMode == MODE_GAME) {
    if (buttonPressed(actionButtonPin, actionBtn)) {
      startDodgeDuel();
    }
    updateGameScreen();
    setRGB(20, 0, 20);
  }

  delay(20);
}
